# [201812-1](http://118.190.20.162/view.page?gpid=T80)

- 遇到红灯，剩多久等多久
- 遇到黄灯，剩多久等多久，再等一个红灯周期
- 遇到绿灯
  - 大于 `0`，直接通过
  - 等于 `0`，等一个黄灯周期 + 红灯周期

## 测试用例

```c++
#include <bits/stdc++.h>
using namespace std;

int ans = 0;

int main() {
	int r, y, g;
	scanf("%d %d %d", &r, &y, &g);
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		int k, t;
		scanf("%d %d", &k, &t);
		if (k == 0) {	// 经过一段道路
			ans += t;
		}
		else if (k == 1) {	// 红灯
			ans += t;	// 等黄灯
		}
		else if (k == 2) {	// 黄灯
			ans += t;	// 等黄灯
			ans += r;	// 等红灯
		}
		else if (k == 3) {	// 绿灯
			if (t == 0) {	// 即将黄灯
				ans += y;	// 等黄灯
				ans += r;	// 等红灯
			}
			else {
				continue;
			}
		}
	}
	printf("%d\n", ans);
	return 0;
}

```

#include <bits/stdc++.h>
using namespace std;

int main() {
	int r, y, g, n, k, t, ans = 0;
	scanf("%d %d %d", &r, &y, &g);
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d %d", &k, &t);
		if (k == 0 || k == 1) {
			ans += t;
		}
		else if (k == 2) {
			ans += t + r;
		}
	}
	printf("%d\n", ans);
	return 0;
}
